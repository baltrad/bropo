88static PyFmiImage*
89PyFmiImage_New(RaveFmiImage_t* p, int width, int height)
90{
91 PyFmiImage* result = NULL;
92 RaveFmiImage_t* cp = NULL;
93
94 if (p == NULL) {
95 cp = RAVE_OBJECT_NEW(&RaveFmiImage_TYPE);
96 if (cp == NULL) {
97 RAVE_CRITICAL0("Failed to allocate memory for fmi image.");
98 raiseException_returnNULL(PyExc_MemoryError, "Failed to allocate memory for fmi image.");
99 }
100 if (width > 0 && height > 0) {
101 if (!RaveFmiImage_initialize(cp, width, height)) {
102 RAVE_CRITICAL0("Failed to initialize fmi image");
103 raiseException_gotoTag(done, PyExc_MemoryError, "Failed to initialize fmi image");
104 }
105 }
106 } else {
107 cp = RAVE_OBJECT_COPY(p);
108 result = RAVE_OBJECT_GETBINDING(p); // If p already have a binding, then this should only be increfed.
109 if (result != NULL) {
110 Py_INCREF(result);
111 }
112 }
113
114 if (result == NULL) {
115 result = PyObject_NEW(PyFmiImage, &PyFmiImage_Type);
116 if (result != NULL) {
117 PYRAVE_DEBUG_OBJECT_CREATED;
118 result->image = RAVE_OBJECT_COPY(cp);
119 RAVE_OBJECT_BIND(result->image, result);
120 } else {
121 RAVE_CRITICAL0("Failed to create PyFmiImage instance");
122 raiseException_gotoTag(done, PyExc_MemoryError, "Failed to allocate memory for fmi image.");
123 }
124 }
125
126done:
127 RAVE_OBJECT_RELEASE(cp);
128 return result;
129}

162static PyObject* _pyfmiimage_fromRave(PyObject* self, PyObject* args)
163{
164 PyObject* inptr = NULL;
165 RaveCoreObject* object = NULL;
166 char* quantity = NULL;
167
168 RaveFmiImage_t* fmiimage = NULL;
169 PyFmiImage* result = NULL;
170
171 if (!PyArg_ParseTuple(args, "O|s", &inptr, &quantity)) {
172 return NULL;
173 }
174
175 if (PyPolarVolume_Check(inptr)) {
176 object = (RaveCoreObject*)PyPolarVolume_GetNative((PyPolarVolume*)inptr);
177 fmiimage = RaveFmiImage_fromPolarVolume((PolarVolume_t*)object, 0, quantity);
178 } else if (PyPolarScan_Check(inptr)) {
179 object = (RaveCoreObject*)PyPolarScan_GetNative((PyPolarScan*)inptr);
180 fmiimage = RaveFmiImage_fromPolarScan((PolarScan_t*)object, quantity);
181 } else if (PyRaveField_Check(inptr)) {
182 object = (RaveCoreObject*)PyRaveField_GetNative((PyRaveField*)inptr);
183 fmiimage = RaveFmiImage_fromRaveField((RaveField_t*)object);
184 } else {
185 raiseException_returnNULL(PyExc_TypeError,"fromRave can handle volumes, scans and rave fields");
186 }
187
188 if (fmiimage != NULL) {
189 result = PyFmiImage_New(fmiimage, 0, 0);
190 } else {
191 PyErr_SetString(PyExc_RuntimeError, "Could not convert rave object into fmiimage");
192 }
193
194 RAVE_OBJECT_RELEASE(fmiimage);
195 RAVE_OBJECT_RELEASE(object);
196
197 return (PyObject*)result;
198}

241static PyObject* _pyfmiimage_addAttribute(PyFmiImage* self, PyObject* args)
242{
243 RaveAttribute_t* attr = NULL;
244 char* name = NULL;
245 PyObject* obj = NULL;
246 PyObject* result = NULL;
247
248 if (!PyArg_ParseTuple(args, "sO", &name, &obj)) {
249 return NULL;
250 }
251
252 attr = RAVE_OBJECT_NEW(&RaveAttribute_TYPE);
253 if (attr == NULL) {
254 return NULL;
255 }
256
257 if (!RaveAttribute_setName(attr, name)) {
258 raiseException_gotoTag(done, PyExc_MemoryError, "Failed to set name");
259 }
260
261 if (PyLong_Check(obj) || PyInt_Check(obj)) {
262 long value = PyLong_AsLong(obj);
263 RaveAttribute_setLong(attr, value);
264 } else if (PyFloat_Check(obj)) {
265 double value = PyFloat_AsDouble(obj);
266 RaveAttribute_setDouble(attr, value);
267 } else if (PyString_Check(obj)) {
268 char* value = PyString_AsString(obj);
269 if (!RaveAttribute_setString(attr, value)) {
270 raiseException_gotoTag(done, PyExc_AttributeError, "Failed to set string value");
271 }
272 } else if (PyArray_Check(obj)) {
273 PyArrayObject* arraydata = (PyArrayObject*)obj;
274 if (PyArray_NDIM(arraydata) != 1) {
275 raiseException_gotoTag(done, PyExc_AttributeError, "Only allowed attribute arrays are 1-dimensional");
276 }
277 if (!RaveAttribute_setArrayFromData(attr, PyArray_DATA(arraydata), PyArray_DIM(arraydata, 0), translate_pyarraytype_to_ravetype(PyArray_TYPE(arraydata)))) {
278 raiseException_gotoTag(done, PyExc_AttributeError, "Failed to set array data");
279 }
280 } else {
281 raiseException_gotoTag(done, PyExc_AttributeError, "Unsupported data type");
282 }
283
284 if (!RaveFmiImage_addAttribute(self->image, attr)) {
285 raiseException_gotoTag(done, PyExc_AttributeError, "Failed to add attribute");
286 }
287
288 result = PyBool_FromLong(1);
289done:
290 RAVE_OBJECT_RELEASE(attr);
291 return result;
292}

300static PyObject* _pyfmiimage_getAttribute(PyFmiImage* self, PyObject* args)
301{
302 RaveAttribute_t* attribute = NULL;
303 char* name = NULL;
304 PyObject* result = NULL;
305 if (!PyArg_ParseTuple(args, "s", &name)) {
306 return NULL;
307 }
308
309 attribute = RaveFmiImage_getAttribute(self->image, name);
310 if (attribute != NULL) {
311 RaveAttribute_Format format = RaveAttribute_getFormat(attribute);
312 if (format == RaveAttribute_Format_Long) {
313 long value = 0;
314 RaveAttribute_getLong(attribute, &value);
315 result = PyLong_FromLong(value);
316 } else if (format == RaveAttribute_Format_Double) {
317 double value = 0.0;
318 RaveAttribute_getDouble(attribute, &value);
319 result = PyFloat_FromDouble(value);
320 } else if (format == RaveAttribute_Format_String) {
321 char* value = NULL;
322 RaveAttribute_getString(attribute, &value);
323 result = PyString_FromString(value);
324 } else if (format == RaveAttribute_Format_LongArray) {
325 long* value = NULL;
326 int len = 0;
327 int i = 0;
328 npy_intp dims[1];
329 RaveAttribute_getLongArray(attribute, &value, &len);
330 dims[0] = len;
331 result = PyArray_SimpleNew(1, dims, PyArray_LONG);
332 for (i = 0; i < len; i++) {
333 *((long*) PyArray_GETPTR1(result, i)) = value[i];
334 }
335 } else if (format == RaveAttribute_Format_DoubleArray) {
336 double* value = NULL;
337 int len = 0;
338 int i = 0;
339 npy_intp dims[1];
340 RaveAttribute_getDoubleArray(attribute, &value, &len);
341 dims[0] = len;
342 result = PyArray_SimpleNew(1, dims, PyArray_DOUBLE);
343 for (i = 0; i < len; i++) {
344 *((double*) PyArray_GETPTR1(result, i)) = value[i];
345 }
346 } else {
347 RAVE_CRITICAL1("Undefined format on requested attribute %s", name);
348 raiseException_gotoTag(done, PyExc_AttributeError, "Undefined attribute");
349 }
350 } else {
351 raiseException_gotoTag(done, PyExc_AttributeError, "No such attribute");
352 }
353done:
354 RAVE_OBJECT_RELEASE(attribute);
355 return result;
356}

364static PyObject* _pyfmiimage_getAttributeNames(PyFmiImage* self, PyObject* args)
365{
366 RaveList_t* list = NULL;
367 PyObject* result = NULL;
368 int n = 0;
369 int i = 0;
370
371 list = RaveFmiImage_getAttributeNames(self->image);
372 if (list == NULL) {
373 raiseException_returnNULL(PyExc_MemoryError, "Could not get attribute names");
374 }
375 n = RaveList_size(list);
376 result = PyList_New(0);
377 for (i = 0; result != NULL && i < n; i++) {
378 char* name = RaveList_get(list, i);
379 if (name != NULL) {
380 PyObject* pynamestr = PyString_FromString(name);
381 if (pynamestr == NULL) {
382 goto fail;
383 }
384 if (PyList_Append(result, pynamestr) != 0) {
385 Py_DECREF(pynamestr);
386 goto fail;
387 }
388 Py_DECREF(pynamestr);
389 }
390 }
391 RaveList_freeAndDestroy(&list);
392 return result;
393fail:
394 RaveList_freeAndDestroy(&list);
395 Py_XDECREF(result);
396 return NULL;
397}
